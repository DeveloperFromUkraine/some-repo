Test Types

This is a high-level description of the different types of tests that are done in relation to Angular.

Unit Tests

Test the JavaScript file in isolation and mock out services as you need them. Unit Tests don't use the Angular Testing Utilities. (*.unit.spec.ts)

            
describe('AppComponent', () => {{ '{' }}
    let component: AppComponent;

    beforeEach(() => {{ '{' }}
        component = new AppComponent();
    });

    it('should be truthy', () => {{ '{' }}
        expect(component).toBeTruthy();
    });
});

Integration Tests

Use the Angular Testing Utilities to generate the Angular environment to test. Check JavaScript integration with the template. Mock out services as you need them. (*.spec.ts)

            
describe('AppComponent', () => {{ '{' }}
    let fixture: ComponentFixture<AppComponent>;
    let component: AppComponent;

    beforeEach(async () => {{ '{' }}
        await TestBed.configureTestingModule({{ '{' }}
            declarations: [AppComponent],
            imports: [],
            providers: []
        }).compileComponents();
    });

    beforeEach(() => {{ '{' }}
        fixture = TestBed.createComponent(AppComponent);
        component = fixture.componentInstance;

        fixture.detectChanges();
    });

    it('should be truthy', () => {{ '{' }}
        expect(component).toBeTruthy();
    });
});

End 2 End Tests

Testing the application as a whole. Simulate a user through real world scenarios. (*.e2e-spec.ts)

            
describe('test-app1 App', () => {{ '{' }}
    let page: AppPage;

    beforeEach(() => {{ '{' }}
        page = new AppPage();
    });

    it('should display welcome message', () => {{ '{' }}
        page.navigateTo();

        expect(page.getParagraphTest()).toEqual('Welcome to app!');
    });
});
