Navigation List

Use

Utilize the nav-list component for navigation lists, such as those with anchor tags. Nav-list components' anchor tags can be wrapped in a ign-list-item for multiple links within the list.

Sample

folder

{{folder.name}}

{{folder.updated | date}}

<ign-nav-list>

    <a ign-list-item onClick=window.open() *ngFor="let folder of folders"

target="_blank" tabindex="0">

        <mat-icon mat-list-icon>folder</mat-icon>

        <h4>{folder.name}</h4>

        <p> {folder.updated | date} </p>

    </a>

</ign-nav-list>

folders = [

  {

    name: 'Click the folders to open a blank page.',

    updated: new Date('1/1/16'),

  },

  {

    name: 'Your descriptor will go here.',

    updated: new Date('1/17/16'),

  },

  {

    name: 'The dates are just for show.',

    updated: new Date('1/28/16'),

  }

];
