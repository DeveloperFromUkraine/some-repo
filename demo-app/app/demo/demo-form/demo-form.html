public formBuilder: FormBuilder;
private close$ = new Subject<void>();
form: FormGroup;
constructor (formBuilder: FormBuilder) {
  this.formBuilder = formBuilder;
}
ngOnInit(): void {
  this.form = this.formBuilder.group({
    myDateRange: this.formBuilder.group({
      startDate: [new Date()],
      endDate: [null]
    })
  })
}
ngOnDestroy(): void {
  this.close$.next();
  this.close$.complete();
}
